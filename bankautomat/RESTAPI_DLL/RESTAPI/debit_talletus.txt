CREATE PROCEDURE debit_talletus(IN first_id INT,IN Summa INT)
  BEGIN
  DECLARE test1 INT DEFAULT 0;
  START TRANSACTION;
  UPDATE debittili SET Saldo=Saldo+Summa WHERE Tilinnumero=first_id;
  SET test1=ROW_COUNT();
    IF (test1 > 0) THEN   
      COMMIT;    
      INSERT INTO tilitapahtumat(Tilinnumero,Tapahtuma,PVM,Summa,CreditTili_Tilinnumero,DebitTili_Tilinnumero) VALUES(first_id,'Talletus',NOW(),Summa,first_id,first_id);
    ELSE
      ROLLBACK;
  END IF;
  END //
